021import ball.annotation.ServiceProviderFor;
022import ball.swing.table.ListTableModel;
023import ball.swing.table.MapTableModel;
024import ball.xml.FluentNode;
025import com.sun.javadoc.ClassDoc;
026import com.sun.javadoc.PackageDoc;
027import com.sun.javadoc.Tag;
028import com.sun.tools.doclets.Taglet;
029import java.io.BufferedReader;
030import java.io.InputStream;
031import java.io.InputStreamReader;
032import java.util.Collection;
033import java.util.Map;
034import java.util.regex.Pattern;
035import lombok.NoArgsConstructor;
036import lombok.ToString;
037
038import static java.nio.charset.StandardCharsets.UTF_8;
039import static java.util.stream.Collectors.joining;
040import static java.util.stream.Collectors.toList;
041import static org.apache.commons.lang3.StringUtils.isNotEmpty;
042
043/**
044 * Inline {@link Taglet} to include a static {@link Class}
045 * {@link java.lang.reflect.Field} or resource in the Javadoc output.
046 *
047 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
048 */
049@TagletName("include")
050@ServiceProviderFor({ Taglet.class })
051@NoArgsConstructor @ToString
052public class IncludeTaglet extends AbstractInlineTaglet implements SunToolsInternalToolkitTaglet {
053    private static final IncludeTaglet INSTANCE = new IncludeTaglet();
054
055    public static void register(Map<Object,Object> map) {
056        register(map, INSTANCE);
057    }
058
059    @Override
060    public FluentNode toNode(Tag tag) throws Throwable {
061        FluentNode node = null;
062        String[] argv = tag.text().trim().split("[\\p{Space}]+", 2);
063
064        if (isNotEmpty(argv[0])) {
065            String[] target = argv[0].trim().split(Pattern.quote("#"), 2);
066
067            if (target.length > 1) {
068                node =
069                    field(tag,
070                          getClassFor(isNotEmpty(target[0]) ? getClassDocFor(tag, target[0]) : containingClass(tag)),
071                          target[1]);
072            } else {
073                Class<?> type = null;
074
075                if (tag.holder() instanceof PackageDoc) {
076                    type = getClassFor((PackageDoc) tag.holder());
077                } else {
078                    type = getClassFor(containingClass(tag));
079                }
080
081                node = resource(tag, type, target[0]);
082            }
083        } else {
084            Class<?> type = getClassFor(containingClass(tag));
085
086            node = field(tag, type, tag.holder().name());
087        }
088
089        return node;
090    }
091
092    private FluentNode field(Tag tag, Class<?> type, String name) throws Exception {
093        FluentNode node = null;
094        Object value = type.getDeclaredField(name).get(null);
095
096        if (value instanceof Collection<?>) {
097            node = table(tag, new ListTableModel(((Collection<?>) value).stream().collect(toList()), "Element"));
098        } else if (value instanceof Map<?,?>) {
099            node = table(tag, new MapTableModel((Map<?,?>) value, "Key", "Value"));
100        } else {
101            node = pre(String.valueOf(value));
102        }
103
104        return div(attr("class", "block"), node);
105    }
106
107    private FluentNode resource(Tag tag, Class<?> type, String name) throws Exception {
108        String string = null;
109
110        if (type == null) {
111            type = getClass();
112        }
113
114        try (InputStream in = type.getResourceAsStream(name)) {
115            string =
116                new BufferedReader(new InputStreamReader(in, UTF_8)).lines()
117                .collect(joining("\n"));
118        }
119
120        return pre(string);
121    }
122}
