021import ball.annotation.ServiceProviderFor;
022import ball.xml.FluentNode;
023import com.sun.javadoc.Tag;
024import com.sun.tools.doclets.Taglet;
025import java.net.URI;
026import java.util.LinkedHashMap;
027import java.util.Map;
028import lombok.NoArgsConstructor;
029import lombok.ToString;
030
031/**
032 * Inline {@link Taglet} to provide external links.
033 *
034 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
035 */
036@TagletName("link.uri")
037@ServiceProviderFor({ Taglet.class })
038@NoArgsConstructor @ToString
039public class LinkURITaglet extends AbstractInlineTaglet implements SunToolsInternalToolkitTaglet {
040    private static final LinkURITaglet INSTANCE = new LinkURITaglet();
041
042    public static void register(Map<Object,Object> map) {
043        register(map, INSTANCE);
044    }
045
046    private static final String SPACES = "[\\p{Space}]+";
047
048    @Override
049    public FluentNode toNode(Tag tag) throws Throwable {
050        String text = tag.text().trim();
051        String[] argv = text.split(SPACES, 2);
052        URI href = new URI(argv[0]);
053
054        text = (argv.length > 1) ? argv[1] : null;
055
056        LinkedHashMap<String,String> map = new LinkedHashMap<>();
057
058        if (text != null) {
059            for (;;) {
060                argv = text.split(SPACES, 2);
061
062                String[] nvp = argv[0].split("=", 2);
063
064                if (argv.length > 1 && nvp.length > 1) {
065                    map.put(nvp[0], nvp[1]);
066                    text = argv[1];
067                } else {
068                    break;
069                }
070            }
071        }
072
073        return a(href, text)
074                   .add(map.entrySet().stream()
075                        .map(t -> attr(t.getKey(), t.getValue())));
076    }
077}
