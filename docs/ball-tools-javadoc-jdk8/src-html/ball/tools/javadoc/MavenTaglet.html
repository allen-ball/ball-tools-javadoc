063/**
064 * Abstract base class for inline {@link Taglet}s that load
065 * {@link.uri https://maven.apache.org/index.html Maven} artifacts.
066 *
067 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
068 */
069@NoArgsConstructor(access = PROTECTED)
070public abstract class MavenTaglet extends AbstractInlineTaglet implements SunToolsInternalToolkitTaglet {
071    private static final XPath XPATH = XPathFactory.newInstance().newXPath();
072
073    private static final String POM_XML = "pom.xml";
074    private static final String DEPENDENCY = "dependency";
075    private static final String GROUP_ID = "groupId";
076    private static final String ARTIFACT_ID = "artifactId";
077    private static final String VERSION = "version";
078
079    protected XPathExpression compile(String format, Object... argv) {
080        XPathExpression expression = null;
081
082        try {
083            expression = XPATH.compile(String.format(format, argv));
084        } catch (Exception exception) {
085            throw new IllegalStateException(exception);
086        }
087
088        return expression;
089    }

091    /**
092     * Method to locate the POM from a {@link Tag}.
093     *
094     * @param   tag             The {@link Tag}.
095     *
096     * @return  The POM {@link File}.
097     *
098     * @throws  Exception       If the POM {@link File} cannot be found.
099     */
100    protected File getPomFileFor(Tag tag) throws Exception {
101        File parent = tag.position().file().getParentFile();
102        String name = defaultIfBlank(tag.text().trim(), POM_XML);
103        File file = new File(parent, name);
104
105        while (parent != null) {
106            file = new File(parent, name);
107
108            if (file.isFile()) {
109                break;
110            } else {
111                file = null;
112            }
113
114            parent = parent.getParentFile();
115        }
116
117        if (! (file != null && file.isFile())) {
118            throw new FileNotFoundException(name);
119        }
120
121        return file;
122    }

124    /**
125     * Inline {@link com.sun.tools.doclets.Taglet} to provide a report of
126     * fields whose values are configured by the
127     * {@link.uri https://maven.apache.org/index.html Maven}
128     * {@link.uri https://maven.apache.org/plugin-developers/index.html Plugin}
129     * {@code plugin.xml}.
130     */
131    @TagletName("maven.plugin.fields")
132    @ServiceProviderFor({ Taglet.class })
133    @NoArgsConstructor @ToString
134    public static class PluginFields extends MavenTaglet {
135        private static final PluginFields INSTANCE = new PluginFields();
136
137        public static void register(Map<Object,Object> map) {
138            register(map, INSTANCE);
139        }
140
141        private static final String PLUGIN_XML = "META-INF/maven/plugin.xml";
142
143        @Override
144        public FluentNode toNode(Tag tag) throws Throwable {
145            ClassDoc doc = null;
146            String[] argv = tag.text().trim().split("[\\p{Space}]+", 2);
147
148            if (isNotEmpty(argv[0])) {
149                doc = getClassDocFor(tag, argv[0]);
150            } else {
151                doc = containingClass(tag);
152            }
153
154            Class<?> type = getClassFor(doc);
155            URL url = getResourceURLOf(type);
156            Protocol protocol = Protocol.of(url);
157            Document document = null;
158
159            switch (protocol) {
160            case FILE:
161                String root = url.getPath().replaceAll(Pattern.quote(getResourcePathOf(type)), EMPTY);
162
163                document =
164                    DocumentBuilderFactory.newInstance()
165                    .newDocumentBuilder()
166                    .parse(root + PLUGIN_XML);
167                break;
168
169            case JAR:
170                JarFile jar = protocol.getJarFile(url);
171                ZipEntry entry = jar.getEntry(PLUGIN_XML);
172
173                try (InputStream in = jar.getInputStream(entry)) {
174                    document =
175                        DocumentBuilderFactory.newInstance()
176                        .newDocumentBuilder()
177                        .parse(in);
178                }
179                break;
180            }
181
182            if (document == null) {
183                throw new IllegalStateException("Cannot find " + PLUGIN_XML);
184            }
185
186            Node mojo =
187                (Node)
188                compile("/plugin/mojos/mojo[implementation='%s']", type.getCanonicalName())
189                .evaluate(document, NODE);
190
191            return div(attr("class", "summary"),
192                       h3("Maven Plugin Parameter Summary"),
193                       table(tag, type, mojo,
194                             asStream((NodeList) compile("parameters/parameter").evaluate(mojo, NODESET))));
195        }
196
197        private FluentNode table(Tag tag, Class<?> type,
198                                 Node mojo, Stream<Node> parameters) {
199            return table(thead(tr(th(EMPTY), th("Field"),
200                                  th("Default"), th("Property"),
201                                  th("Required"), th("Editable"),
202                                  th("Description"))),
203                         tbody(parameters.map(t -> tr(tag, type, mojo, t))));
204        }
205
206        private FluentNode tr(Tag tag, Class<?> type, Node mojo, Node parameter) {
207            FluentNode tr = fragment();
208
209            try {
210                String name = compile("name").evaluate(parameter);
211                Field field = FieldUtils.getField(type, name, true);
212
213                if (field != null) {
214                    tr =
215                        tr(td((! type.equals(field.getDeclaringClass()))
216                                  ? type(tag, field.getDeclaringClass())
217                                  : text(EMPTY)),
218                           td(declaration(tag, field)),
219                           td(code(compile("configuration/%s/@default-value", name).evaluate(mojo))),
220                           td(code(compile("configuration/%s", name).evaluate(mojo))),
221                           td(code(compile("required").evaluate(parameter))),
222                           td(code(compile("editable").evaluate(parameter))),
223                           td(p(compile("description").evaluate(parameter))));
224                }
225            } catch (RuntimeException exception) {
226                throw exception;
227            } catch (Exception exception) {
228                throw new IllegalStateException(exception);
229            }
230
231            return tr;
232        }
233    }

235    /**
236     * Inline {@link com.sun.tools.doclets.Taglet} to include generated
237     * {@link.uri https://maven.apache.org/index.html Maven}
238     * {@link.uri https://maven.apache.org/plugin-developers/index.html Plugin}
239     * help documentation.
240     */
241    @TagletName("maven.plugin.help")
242    @ServiceProviderFor({ Taglet.class })
243    @NoArgsConstructor @ToString
244    public static class PluginHelp extends MavenTaglet {
245        private static final PluginHelp INSTANCE = new PluginHelp();
246
247        public static void register(Map<Object,Object> map) {
248            register(map, INSTANCE);
249        }
250
251        private static final String NAME = "plugin-help.xml";
252        @CompileTimeCheck
253        private static final Pattern PATTERN =
254            Pattern.compile("META-INF/maven/(?<g>[^/]+)/(?<a>[^/]+)/" + Pattern.quote(NAME));
255
256        @Override
257        public FluentNode toNode(Tag tag) throws Throwable {
258            Class<?> type = null;
259
260            if (tag.holder() instanceof PackageDoc) {
261                type = getClassFor((PackageDoc) tag.holder());
262            } else {
263                type = getClassFor(containingClass(tag));
264            }
265
266            URL url = getResourceURLOf(type);
267            Protocol protocol = Protocol.of(url);
268            Document document = null;
269
270            switch (protocol) {
271            case FILE:
272                Path root = Paths.get(url.getPath().replaceAll(Pattern.quote(getResourcePathOf(type)), EMPTY));
273                Path path =
274                    Files.walk(root, Integer.MAX_VALUE)
275                    .filter(Files::isRegularFile)
276                    .filter(t -> PATTERN.matcher(root.relativize(t).toString()).matches())
277                    .findFirst().orElse(null);
278
279                document =
280                    DocumentBuilderFactory.newInstance()
281                    .newDocumentBuilder()
282                    .parse(path.toFile());
283                break;
284
285            case JAR:
286                try (JarFile jar = protocol.getJarFile(url)) {
287                    JarEntry entry =
288                        jar.stream()
289                        .filter(t -> PATTERN.matcher(t.getName()).matches())
290                        .findFirst().orElse(null);
291
292                    try (InputStream in = jar.getInputStream(entry)) {
293                        document =
294                            DocumentBuilderFactory.newInstance()
295                            .newDocumentBuilder()
296                            .parse(in);
297                    }
298                }
299                break;
300            }
301
302            if (document == null) {
303                throw new IllegalStateException("Cannot find " + NAME);
304            }
305
306            return div(attr("class", "summary"),
307                       h3(compile("/plugin/name").evaluate(document)),
308                       p(compile("/plugin/description").evaluate(document)),
309                       table(tag, asStream((NodeList) compile("/plugin/mojos/mojo").evaluate(document, NODESET))));
310        }
311
312        private FluentNode table(Tag tag, Stream<Node> mojos) {
313            return table(thead(tr(th("Goal"), th("Phase"), th("Description"))),
314                         tbody(mojos.map(t -> tr(tag, t))));
315        }
316
317        private FluentNode tr(Tag tag, Node mojo) {
318            FluentNode tr = fragment();
319
320            try {
321                tr =
322                    tr(td(a(tag, compile("implementation").evaluate(mojo), code(compile("goal").evaluate(mojo)))),
323                       td(code(compile("phase").evaluate(mojo))),
324                       td(p(code(compile("description").evaluate(mojo)))));
325            } catch (RuntimeException exception) {
326                throw exception;
327            } catch (Exception exception) {
328                throw new IllegalStateException(exception);
329            }
330
331            return tr;
332        }
333    }

335    /**
336     * {@link com.sun.tools.doclets.Taglet} to provide
337     * {@link.uri https://maven.apache.org/pom.html POM} coordinates as
338     * a {@code <dependency/>} element to include this documented
339     * {@link Class} or {@link Package}.
340     *
341     * <p>For example:</p>
342     *
343     * {@pom.coordinates}
344     */
345    @TagletName("pom.coordinates")
346    @ServiceProviderFor({ Taglet.class })
347    @NoArgsConstructor @ToString
348    public static class Coordinates extends MavenTaglet {
349        private static final Coordinates INSTANCE = new Coordinates();
350
351        public static void register(Map<Object,Object> map) {
352            register(map, INSTANCE);
353        }
354
355        @CompileTimeCheck
356        private static final Pattern PATTERN =
357            Pattern.compile("META-INF/maven/(?<g>[^/]+)/(?<a>[^/]+)/pom[.]properties");
358
359        @Override
360        public FluentNode toNode(Tag tag) throws Throwable {
361            POMProperties properties = new POMProperties();
362            Class<?> type = null;
363
364            if (tag.holder() instanceof PackageDoc) {
365                type = getClassFor((PackageDoc) tag.holder());
366            } else {
367                type = getClassFor(containingClass(tag));
368            }
369
370            URL url = getResourceURLOf(type);
371            Protocol protocol = Protocol.of(url);
372
373            switch (protocol) {
374            case FILE:
375                Document document =
376                    DocumentBuilderFactory.newInstance()
377                    .newDocumentBuilder()
378                    .parse(getPomFileFor(tag));
379
380                Stream.of(GROUP_ID, ARTIFACT_ID, VERSION)
381                    .forEach(t -> properties.load(t, document, "/project/"));
382                Stream.of(VERSION)
383                    .forEach(t -> properties.load(t, document, "/project/parent/"));
384                break;
385
386            case JAR:
387                try (JarFile jar = protocol.getJarFile(url)) {
388                    JarEntry entry =
389                        jar.stream()
390                        .filter(t -> PATTERN.matcher(t.getName()).matches())
391                        .findFirst().orElse(null);
392
393                    if (entry != null) {
394                        try (InputStream in = jar.getInputStream(entry)) {
395                            properties.load(in);
396                        }
397                    }
398                }
399                break;
400            }
401
402            return pre("xml",
403                       render(element(DEPENDENCY,
404                                      Stream.of(GROUP_ID, ARTIFACT_ID, VERSION)
405                                      .map(t -> element(t).content(properties.getProperty(t, "unknown")))),
406                              2));
407        }
408    }
409
410    private static enum Protocol {
411        FILE, JAR;
412
413        public static Protocol of(URL url) {
414            return valueOf(url.getProtocol().toUpperCase());
415        }
416
417        public JarFile getJarFile(URL url) throws IOException {
418            JarFile jar = null;
419
420            switch (this) {
421            case JAR:
422                jar = ((JarURLConnection) url.openConnection()).getJarFile();
423                break;
424
425            default:
426                throw new IllegalStateException();
427                /* break; */
428            }
429
430            return jar;
431        }
432    }
433
434    private static class POMProperties extends PropertiesImpl {
435        private static final long serialVersionUID = 1354153174028868013L;
436
437        public String load(String key, Document document, String prefix) {
438            if (document != null) {
439                computeIfAbsent(key, k -> evaluate(prefix + k, document));
440            }
441
442            return super.getProperty(key);
443        }
444
445        private String evaluate(String expression, Document document) {
446            String value = null;
447
448            try {
449                value = XPATH.evaluate(expression, document);
450            } catch (Exception exception) {
451            }
452
453            return isNotEmpty(value) ? value : null;
454        }
455    }
