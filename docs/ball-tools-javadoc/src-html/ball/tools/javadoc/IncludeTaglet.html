021import ball.annotation.ServiceProviderFor;
022import ball.swing.table.ListTableModel;
023import ball.swing.table.MapTableModel;
024import ball.xml.FluentNode;
025import com.sun.source.doctree.UnknownInlineTagTree;
026import java.io.BufferedReader;
027import java.io.InputStreamReader;
028import java.lang.reflect.Field;
029import java.util.Collection;
030import java.util.Map;
031import java.util.regex.Pattern;
032import javax.lang.model.element.Element;
033import javax.lang.model.element.PackageElement;
034import javax.lang.model.element.TypeElement;
035import javax.lang.model.element.VariableElement;
036import jdk.javadoc.doclet.Taglet;
037import lombok.NoArgsConstructor;
038import lombok.ToString;
039
040import static java.nio.charset.StandardCharsets.UTF_8;
041import static java.util.stream.Collectors.joining;
042import static java.util.stream.Collectors.toList;
043import static javax.lang.model.util.ElementFilter.fieldsIn;
044import static org.apache.commons.lang3.StringUtils.isNotEmpty;
045
046/**
047 * Inline {@link Taglet} to include a static {@link Class}
048 * {@link java.lang.reflect.Field} or resource in the Javadoc output.
049 *
050 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
051 */
052@TagletName("include")
053@ServiceProviderFor({ Taglet.class })
054@NoArgsConstructor @ToString
055public class IncludeTaglet extends AbstractInlineTaglet {
056    @Override
057    public FluentNode toNode(UnknownInlineTagTree tag, Element context) throws Throwable {
058        FluentNode node = null;
059        var argv = getText(tag).trim().split("[\\p{Space}]+", 2);
060
061        if (isNotEmpty(argv[0])) {
062            var target = argv[0].trim().split(Pattern.quote("#"), 2);
063
064            if (target.length > 1) {
065                var type =
066                    isNotEmpty(target[0])
067                        ? getTypeElementFor(context, target[0])
068                        : getEnclosingTypeElement(context);
069                var field =
070                    fieldsIn(type.getEnclosedElements()).stream()
071                    .filter(t -> t.getSimpleName().contentEquals(target[1]))
072                    .findFirst().get();
073
074                node = field(tag, context, field);
075            } else {
076                var type =
077                    (context instanceof PackageElement)
078                        ? asPackageInfoClass((PackageElement) context)
079                        : asClass(getEnclosingTypeElement(context));
080
081                node = resource(tag, context, type, target[0]);
082            }
083        } else {
084            node = field(tag, context, (VariableElement) context);
085        }
086
087        return node;
088    }
089
090    private FluentNode field(UnknownInlineTagTree tag, Element context, VariableElement element) throws Exception {
091        FluentNode node = null;
092        var type = asClass((TypeElement) element.getEnclosingElement());
093        var field = type.getDeclaredField(element.getSimpleName().toString());
094
095        field.setAccessible(true);
096
097        var value = field.get(null);
098
099        if (value instanceof Collection<?>) {
100            node =
101                table(tag, context, new ListTableModel(((Collection<?>) value).stream().collect(toList()), "Element"));
102        } else if (value instanceof Map<?,?>) {
103            node = table(tag, context, new MapTableModel((Map<?,?>) value, "Key", "Value"));
104        } else {
105            node = pre(String.valueOf(value));
106        }
107
108        return div(attr("class", "block"), node);
109    }
110
111    private FluentNode resource(UnknownInlineTagTree tag, Element context, Class<?> type, String name) throws Exception {
112        String string = null;
113
114        if (type == null) {
115            type = getClass();
116        }
117
118        try (var in = type.getResourceAsStream(name)) {
119            string =
120                new BufferedReader(new InputStreamReader(in, UTF_8)).lines()
121                .collect(joining("\n"));
122        }
123
124        return pre(string);
125    }
126}
